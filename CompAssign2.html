#Question 2
import matplotlib.pyplot as plt

r= numpy.linspace(1, 5) #number of samples starting domain at 1 because must begin at number>0
def V(r):  #defining function V (intermolecular potential between neutral atoms) which is dependant on variable r (distance between neutral atoms)
    rm = 2**(1/6) #defining constant, (minimum of the potential well)
    return ((rm/r)**12-(2*(rm/r)**6)) #this is the function V(r) to be used (returned)

plt.plot(r, V(r)) #plotting a graph of V vs r 
plt.xlabel("Distance between neutral atoms  (r)") #labelling axis
plt.ylabel("Intermolecular potential between neutral atoms (V)")
plt.xlim(1,5) #chose scale of 5 opposed to number>5 to get a close accurate look at whats happening in the graph 
plt.ylim(-1.5, 0.1) #chose small scale that goes negative to catch whole shape of curve
plt.title("Lennard-Jones Potential") #title
plt.savefig("compassign2_Lennard-Jones_potential_fig.pdf") #saving figure of graph

from numpy import pi #importing symbol constants that will be used later in function
from numpy import e

vo = numpy.arange(-100,100) #This is the range that works for the data to show the full probability distribution curves

def F(vo,T): #defining function F and the variables it's dependant on; voltage (vo) and tempature (T)
    k = 1.38*10**(-23) #defining constant variables in function
    m = 4.652*10**(-23)
    return (m/(2*pi*k*T)**(3/2))*4*pi*vo**2*e**(-m*vo**2/(2*k*T)) #this is the function F(vo, T) to be used (returned)

plt.plot(vo,F(vo,80), label="Tempature of 80K") #plotting vo(voltage) against The probability distribution for vo and a specific tempature, plotting all other tempatures from lowest to highest follows
plt.plot(vo,F(vo,233), label="Tempature of 233K")
plt.plot(vo,F(vo,298), label="Tempature of 298K") 
plt.xlabel("The speed of nitrogen particles (m/s)") #labelling axis
plt.ylabel("The probability distribution")
plt.title("Maxwell-Boltzmann distribution for varying temperatures") #title 
plt.legend() #legend for labels of different tempature curves
plt.savefig("CompAssign2_Maxwell-Boltzman_distribution_varying_temps_fig.pdf") #saving figure of graph
